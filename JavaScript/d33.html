class ShapeCalculator { constructor(shape) { this.shape = shape; } calculateArea() { switch (this.shape) { case 'triangle': const base = parseFloat(prompt('Enter base:')); const height = parseFloat(prompt('Enter height:')); return 0.5 * base * height; case 'rectangle': const length = parseFloat(prompt('Enter length:')); const width = parseFloat(prompt('Enter width:')); return length * width; case 'square': const side = parseFloat(prompt('Enter side:')); return side * side; default: return 'Invalid shape'; } } } const prompt = require("prompt-sync")() const shape = prompt('Enter shape (triangle, rectangle, square):'); const calculator = new ShapeCalculator(shape); const area = calculator.calculateArea(); console.log(`Area: ${area}`);
